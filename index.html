ARM Assembly Programs

1b. Factorial Program


AREA FACTORIAL, CODE, READONLY
ENTRY

START
    MOV R0, #4
    MOV R1, R0

FACT
    SUBS R1, R1, #1
    CMP R1, #1
    BEQ STOP
    MUL R3, R0, R1
    MOV R0, R3
    B FACT

STOP
    B STOP

END

1c. Add Array Program


AREA ADDITION, CODE, READONLY
ENTRY

START
    MOV R5, #6
    MOV R0, #0
    LDR R1, =VALUE1

LOOP
    LDRH R2, [R1], #2
    LDR R3, MASK
    AND R2, R2, R3
    ADD R0, R0, R2
    SUBS R5, R5, #1
    BNE LOOP

    LDR R4, =RESULT
    STR R0, [R4]

XSS
    B XSS

MASK    DCD 0x0000FFFF
VALUE1  DCW 0x1111, 0x2222, 0x3333, 0xAAAA, 0xBBBB, 0xCCCC

AREA DATA2, DATA, READWRITE
RESULT  DCD 0

END
